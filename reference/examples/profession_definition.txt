-- Build 42 profession definition example
-- Place at: media/scripts/characters/myprofessions.txt

module Base
{
    -- ── A custom profession ────────────────────────────────────────────────────
    character_profession_definition base:mymod_scavenger
    {
        CharacterProfession = base:mymod_scavenger,

        -- Negative = player earns points back (profession is a disadvantage overall)
        -- Positive = player spends points (it's a strong profession)
        -- 0        = neutral
        Cost = -4,

        UIName        = UI_prof_MyMod_Scavenger,     -- or a literal string: "Scavenger"
        UIDescription = UI_profdesc_MyMod_Scavenger, -- or a literal string
        IconPathName  = profession_mymod_scavenger,   -- file: media/textures/profession_mymod_scavenger.png

        -- Free profession-locked trait granted at character start
        -- Must be a trait with IsProfessionTrait = true (see mytraits.txt)
        GrantedTraits = base:mymod_scrounger_prof,

        -- Skill XP boosts granted at creation (semicolon-separated)
        XPBoosts = Maintenance=2;SmallBlade=1;Lightfoot=1,

        -- Recipes unlocked at character creation (semicolon-separated recipe names)
        GrantedRecipes = MakeShiv;MakeGlassShiv;MakeToothbrushShiv,
    }
}
